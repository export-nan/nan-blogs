                            # еҲқе§ӢеҢ–йЎ№зӣ®

                            ж–°е»әдёҖдёӘж–Үд»¶еӨ№дҪңдёәйЎ№зӣ®ж №зӣ®еҪ•пјҢиҝӣе…ҘиҜҘж–Үд»¶еӨ№пјҢжү§иЎҢд»ҘдёӢе‘Ҫд»Ө

                            go mod init test

                            еңЁиҝҷйҮҢжҲ‘д»¬е°ҶеҪ“еүҚйЎ№зӣ®зҡ„еҗҚз§°е®ҡдёәtest,
                            дёҠйқўзҡ„д»Јз ҒдјҡеңЁеҪ“еүҚж–Үд»¶еӨ№дёӢз”ҹжҲҗдёҖдёӘ go.mod ж–Үд»¶пјҢиҜҘж–Үд»¶з”ЁдәҺз®ЎзҗҶйЎ№зӣ®дҫқиө–гҖӮ

                            ./
в””в”Җв”Җ go.mod

                            еңЁgo.mod ж–Үд»¶дёӯпјҢжҲ‘д»¬иғҪзңӢеҲ°д»ҘдёӢеҶ…е®№пјҡ

                            module test // жҢҮе®ҡеҪ“еүҚйЎ№зӣ®зҡ„еҗҚз§°

go 1.20 // жҢҮе®ҡеҪ“еүҚйЎ№зӣ®зҡ„ Go зүҲжң¬

                            # еҲӣе»әдёҖдёӘеҢ…

                            еңЁGoдёӯжҲ‘д»¬йҖҡеёёз”Ёж–Үд»¶жҲ–ж–Үд»¶зӣ®еҪ•жқҘиЎЁзӨәдёҖдёӘеҢ…пјҢжҜ”еҰӮжҲ‘д»¬еңЁйЎ№зӣ®ж №зӣ®еҪ•дёӢж–°е»әдёҖдёӘе‘ҪеҗҚдёәcalculatorзҡ„зӣ®еҪ•пјҢжҲ‘д»¬еҸҜд»Ҙе°ҶеҪ“еүҚиҝҷдёӘзӣ®еҪ•зңӢеҒҡдёәдёҖдёӘеҢ…пјҢеҸҠжҲ‘д»¬е»әз«ӢдәҶдёҖдёӘеҢ…еҗҚдёәcalculatorзҡ„еҢ…

                            ./
в”ңв”Җв”Җ calculator (d) // calculatorеҢ…
в””в”Җв”Җ go.mod

                            еңЁGoеҢ…дёӯжҲ‘д»¬еҸҜиғҪжңүеҫҲеӨҡе…·дҪ“еҠҹиғҪзҡ„жЁЎеқ—, жҲ‘д»¬еҸҜд»Ҙз”Ёж–Үд»¶жқҘиЎЁзӨәд»–д»¬пјҢеңЁcalculatorзӣ®еҪ•дёӢж–°е»әдёҖдёӘж–Үд»¶пјҢе‘ҪеҗҚдёәlogic.goзҡ„ж–Үд»¶пјҢеҸҠжҲ‘д»¬еңЁcalculatorеҢ…дёӢж–°е»әдәҶдёҖдёӘеҗҚдёәlogicзҡ„жЁЎеқ—

                            ./
в”ңв”Җв”Җ calculator (d) // calculatorеҢ…

в””в”Җв”Җ go.mod

                            еңЁlogic.goж–Үд»¶дёӯпјҢжҲ‘д»¬еҶҷе…Ҙд»ҘдёӢеҶ…е®№пјҡ

                            package calculator // иЎЁзӨәеҪ“еүҚжЁЎеқ—еұһдәҺcalculatorеҢ…

/*
  1. func -> еЈ°жҳҺеҮҪж•°пјҲfunctionпјүзҡ„е…ій”®еӯ—
  2. Add -> еҮҪж•°еҗҚ, еңЁGoиҜӯиЁҖдёӯйңҖиҰҒиў«е…¬е…ұи°ғз”Ёзҡ„ж–№жі•йңҖиҰҒеӨ§й©јеі°е‘ҪеҗҚпјҢеҸҜд»ҘзҗҶи§Јдёә public Add
  3. зұ»еһӢе®ҡд№ү -> GoиҜӯиЁҖдёӯзұ»еһӢе®ҡд№үж”ҫеңЁеҸҳйҮҸеҗҺйқў
  4. (a, b int) -> еҮҪж•°зҡ„еҸӮж•°еҲ—иЎЁпјҢеӨҡдёӘеҸӮж•°д№Ӣй—ҙз”ЁйҖ—еҸ·йҡ”ејҖпјҢ еңЁи°ғз”Ёж—¶йңҖиҰҒжҢүз…§йЎәеәҸдј е…Ҙ
  5. int -> еҮҪж•°зҡ„иҝ”еӣһеҖјзұ»еһӢпјҢGoиҜӯиЁҖдёӯеҮҪж•°еҸӘиғҪжңүдёҖдёӘиҝ”еӣһеҖјпјҢеӨҡдёӘиҝ”еӣһеҖјйңҖиҰҒз”ЁжӢ¬еҸ·еҢ…иЈ№иө·жқҘ
*/

func Add(a int, b int) int {
  return a + b
}
